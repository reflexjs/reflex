export const Posts = ({ posts }) => {
  return posts.length ? (
    <Grid col="1" gap="10|20" maxW="800px" mx="auto">
      {posts &&
        posts.map((post, index) => (
          <div key={index}>
            <PostTeaser post={post} />
          </div>
        ))}
    </Grid>
  ) : null
}

export const PostTeaser = ({
  post: { title, excerpt, slug, image, date, datetime, author, timeToRead },
  ...props
}) => (
  <article {...props}>
    <div mt="4">
      {title && (
        <Link href={slug}>
          <h2
            variant="heading.h2"
            mt="0"
            mb="4"
            fontSize="2xl|2xl|3xl|4xl"
            _hover={{
              color: "primary",
            }}
          >
            {title}
          </h2>
        </Link>
      )}
      <PostMeta
        author={author}
        timeToRead={timeToRead}
        date={date}
        datetime={datetime}
        fontSize="md"
      />
      {image && (
        <Link href={slug} display="block" mt="4">
          <Image src={image} title={title} alt={title} />
        </Link>
      )}
      {excerpt && <p mt="4">{excerpt}</p>}
    </div>
  </article>
)
